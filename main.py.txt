# ============ مدل‌های داده ============
class UserCreate(BaseModel):
    username: str
    volume_gb: float
    expire_days: int = 30
    protocol: str = "vless"
    inbound_tag: Optional[str] = None
    status: str = "active"
    note: Optional[str] = None

class UserUpdate(BaseModel):
    volume_gb: Optional[float] = None
    expire_days: Optional[int] = None

# ============ توابع کمکی ============
def get_user_info(username: str) -> dict:
    """دریافت اطلاعات یک کاربر از پنل"""
    try:
        response = requests.get(
            f"{PANEL_URL}/api/user/{username}",
            headers=headers,
            timeout=10
        )
        if response.status_code == 404:
            raise HTTPException(status_code=404, detail="User not found")
        if response.status_code != 200:
            raise HTTPException(status_code=response.status_code, detail=response.text)
        return response.json()
    except requests.exceptions.RequestException as e:
        raise HTTPException(status_code=500, detail=f"Connection error: {str(e)}")

def get_inbounds() -> list:
    """دریافت لیست اینباندها از پنل"""
    try:
        response = requests.get(
            f"{PANEL_URL}/api/inbounds",
            headers=headers,
            timeout=10
        )
        if response.status_code == 200:
            return response.json()
        return []
    except:
        return []

def get_protocol_config(protocol: str) -> dict:
    """دریافت کانفیگ مخصوص هر پروتکل"""
    configs = {
        "vless": {"flow": "xtls-rprx-vision", "encryption": "none"},
        "vmesh": {"security": "auto", "alterId": 0},
        "trojan": {"flow": "xtls-rprx-vision"},
        "shadowsocks": {"method": "chacha20-ietf-poly1305"}
    }
    return configs.get(protocol, configs["vless"])

def ping_user(username: str) -> dict:
    """گرفتن پینگ از کاربر از طریق نود"""
    try:
        user_info = get_user_info(username)
        response = requests.get(
            f"{NODE_URL}/api/status",
            params={"username": username},
            timeout=5
        )
        if response.status_code == 200:
            data = response.json()
            ping_ms = data.get('ping', -1)
            if ping_ms > 0 and ping_ms < 1000:
                status = "online"
            elif ping_ms >= 1000:
                status = "high_latency"
            else:
                status = "offline"
            return {
                "username": username,
                "ping_ms": ping_ms,
                "status": status,
                "timestamp": datetime.now().isoformat()
            }
        else:
            return {
                "username": username,
                "ping_ms": -1,
                "status": "unknown",
                "timestamp": datetime.now().isoformat()
            }
    except:
        return {
            "username": username,
            "ping_ms": -1,
            "status": "unreachable",
            "timestamp": datetime.now().isoformat()
        }

@app.post("/api/user/create", status_code=201)
async def create_user(user: UserCreate):
    """
    ایجاد کاربر جدید با حجم مشخص
    
    - username: نام کاربری (حداقل 3 کاراکتر)
    - volume_gb: حجم به گیگابایت
    - expire_days: مدت اعتبار به روز
    - protocol: vless, vmesh, trojan, shadowsocks
    """
    # اعتبارسنجی
    if len(user.username) < 3:
        raise HTTPException(status_code=400, detail="Username must be at least 3 characters")
    
    if user.volume_gb <= 0:
        raise HTTPException(status_code=400, detail="Volume must be greater than 0")
    
    valid_protocols = ["vless", "vmesh", "trojan", "shadowsocks"]
    if user.protocol not in valid_protocols:
        raise HTTPException(status_code=400, detail=f"Invalid protocol. Choose from: {valid_protocols}")
    
    # بررسی وجود کاربر
    try:
        existing = get_user_info(user.username)
        if existing:
            raise HTTPException(status_code=400, detail="Username already exists")
    except HTTPException as e:
        if e.status_code != 404:
            raise e
    
    # پیدا کردن اینباند مناسب
    inbounds = get_inbounds()
    if user.inbound_tag:
        inbound_tag = user.inbound_tag
    else:
        matching_inbounds = [i for i in inbounds if i.get('protocol', '').lower() == user.protocol]
        if matching_inbounds:
            inbound_tag = matching_inbounds[0].get('tag')
        else:
            inbound_tag = f"{user.protocol.upper()}-TCP-XTLS"
    
    # محاسبه تاریخ انقضا
    expire_date = datetime.now() + timedelta(days=user.expire_days)
    expire_timestamp = int(expire_date.timestamp())
    proxy_config = get_protocol_config(user.protocol)
    
    # ساخت داده برای ارسال به پنل
    data = {
        "username": user.username,
        "proxies": {
            user.protocol: proxy_config
        },
        "inbounds": {
            user.protocol: [inbound_tag]
        },
        "expire": expire_timestamp,
        "data_limit": int(user.volume_gb * 1024**3),
        "status": user.status
    }
    
    if user.note:
        data["note"] = user.note
        try:
        response = requests.post(
            f"{PANEL_URL}/api/user",
            json=data,
            headers=headers,
            timeout=10
        )
        if response.status_code in [200, 201]:
            result = response.json()
            sub_url = f"{PANEL_URL}/api/user/{user.username}/subscription?sub=1"
            
            return {
                "success": True,
                "message": f"User {user.username} created successfully",
                "user": result,
                "subscription_url": sub_url,
                "config": {
                    "username": user.username,
                    "volume": f"{user.volume_gb} GB",
                    "expire": expire_date.isoformat(),
                    "protocol": user.protocol,
                    "inbound": inbound_tag
                }
            }
        else:
            raise HTTPException(status_code=response.status_code, detail=response.text)
    except requests.exceptions.RequestException as e:
        raise HTTPException(status_code=500, detail=f"Connection error: {str(e)}")

@app.get("/api/user/{username}/subscription")
async def get_subscription(username: str, format: str = Query("json", enum=["json", "base64"])):
    """دریافت لینک اشتراک بدون فیلتر"""
    user_info = get_user_info(username)
    sub_url = f"{PANEL_URL}/api/user/{username}/subscription?sub=1"
    
    content = None
    if format == "base64":
        try:
            response = requests.get(sub_url, timeout=10)
            if response.status_code == 200:
                content = response.text
        except:
            pass
    
    used = user_info.get('used_traffic', 0)
    limit = user_info.get('data_limit', 0)
    expire = user_info.get('expire', 0)
    
    return {
        "success": True,
        "username": username,
        "subscription_url": sub_url,
        "raw_config": content,
        "user_info": {
            "used_gb": round(used / 1024**3, 2),
            "limit_gb": round(limit / 1024**3, 2) if limit else "Unlimited",
            "remaining_gb": round((limit - used) / 1024**3, 2) if limit else "Unlimited",
            "usage_percent": round((used / limit * 100), 1) if limit else 0,
            "status": user_info.get('status', 'unknown'),
            "expire": datetime.fromtimestamp(expire).isoformat() if expire else "Never"
        }
    }

@app.get("/api/users/list")

@app.get("/api/statistics/total")
async def get_statistics():
    """دریافت آمار کلی پنل"""
    try:
        response = requests.get(
            f"{PANEL_URL}/api/users",
            headers=headers,
            timeout=10
        )
        
        if response.status_code == 200:
            users = response.json()
            total = len(users)
            active = sum(1 for u in users if u.get('status') == 'active')
            total_used = sum(u.get('used_traffic', 0) for u in users)
            total_limit = sum(u.get('data_limit', 0) for u in users)
            
            return {
                "success": True,
                "statistics": {
                    "total_users": total,
                    "active_users": active,
                    "inactive_users": total - active,
                    "total_used_gb": round(total_used / 1024**3, 2),
                    "total_limit_gb": round(total_limit / 1024**3, 2) if total_limit else 0
                }
            }
        else:
            raise HTTPException(status_code=response.status_code, detail=response.text)
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.put("/api/user/{username}/update")
async def update_user(username: str, update_data: UserUpdate):
    """به‌روزرسانی اطلاعات کاربر"""
    try:
        current = get_user_info(username)
        update_payload = {}
        
        if update_data.volume_gb is not None:
            if update_data.volume_gb <= 0:
                raise HTTPException(status_code=400, detail="Volume must be greater than 0")
            update_payload["data_limit"] = int(update_data.volume_gb * 1024**3)
        
        if update_data.expire_days is not None:
            if update_data.expire_days <= 0:
                raise HTTPException(status_code=400, detail="Expire days must be greater than 0")
            expire_date = datetime.now() + timedelta(days=update_data.expire_days)
            update_payload["expire"] = int(expire_date.timestamp())
        
        if update_data.status is not None:
            if update_data.status not in ["active", "disabled", "limited"]:
                raise HTTPException(status_code=400, detail="Invalid status")
            update_payload["status"] = update_data.status
        
        if update_data.note is not None:
            update_payload["note"] = update_data.note
        
        if not update_payload:
            raise HTTPException(status_code=400, detail="No fields to update")
        
        response = requests.put(
            f"{PANEL_URL}/api/user/{username}",
            json=update_payload,
            headers=headers,
            timeout=10
        )
        
        if response.status_code == 200:
            return {
                "success": True,
                "message": f"User {username} updated successfully",
                "updated_fields": update_payload
            }
        else:
            raise HTTPException(status_code=response.status_code, detail=response.text)
    except HTTPException:
        raise
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.exception_handler(HTTPException)
async def http_exception_handler(request, exc):
    return JSONResponse(
        status_code=exc.status_code,
        content={
            "success": False,
            "error": exc.detail,
            "status_code": exc.status_code,
            "timestamp": datetime.now().isoformat()
        }
    )

@app.exception_handler(Exception)
async def general_exception_handler(request, exc):
    logger.error(f"Unexpected error: {str(exc)}")
    return JSONResponse(
        status_code=500,
        content={
            "success": False,
            "error": "Internal server error",
            "status_code": 500,
            "timestamp": datetime.now().isoformat()
        }
    )
